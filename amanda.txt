    <script>
        let vendas = []; // Array vazio que será preenchido pela API


        // Toggle sidebar
        document.getElementById('toggle-sidebar').addEventListener('click', function () {
            document.querySelector('.sidebar').classList.toggle('collapsed');
            document.querySelector('.main-content').classList.toggle('ml-64');
            document.querySelector('.main-content').classList.toggle('ml-20');

            const icon = this.querySelector('i');
            if (document.querySelector('.sidebar').classList.contains('collapsed')) {
                icon.classList.remove('fa-chevron-left');
                icon.classList.add('fa-chevron-right');
            } else {
                icon.classList.remove('fa-chevron-right');
                icon.classList.add('fa-chevron-left');
            }
        });

        // Initialize page
        document.addEventListener('DOMContentLoaded', function () {
            carregarVendas();

            // Set default date/time
            const now = new Date();
            const formattedDateTime = now.toISOString().slice(0, 16);
            document.getElementById('venda-data').value = formattedDateTime;
        });

        async function carregarVendas() {
            try {
                console.log('Iniciando carregamento de vendas...');

                // Carrega as vendas
                const response = await fetch('/api/sales');
                if (!response.ok) {
                    throw new Error(`Erro HTTP: ${response.status}`);
                }

                const vendas = await response.json();
                console.log('Dados recebidos:', vendas);

                // Garante que vendas seja um array
                const dadosVendas = Array.isArray(vendas) ? vendas : [];
                console.log('Dados formatados:', dadosVendas);

                renderizarTabelaVendas(dadosVendas);

                // Carrega o resumo
                const summaryResponse = await fetch('/api/sales/summary');
                if (!summaryResponse.ok) {
                    throw new Error(`Erro HTTP: ${summaryResponse.status}`);
                }

                const summary = await summaryResponse.json();
                console.log('Resumo recebido:', summary);

                // Garante valores padrão
                atualizarResumo({
                    today: summary?.today || 0,
                    week: summary?.week || 0,
                    month: summary?.month || 0,
                    pending: summary?.pending || 0
                });
                if (summary) {
                    document.getElementById('vendas-hoje').textContent = `R$ ${summary.today.toFixed(2).replace('.', ',')}`;
                    document.getElementById('vendas-semana').textContent = `R$ ${summary.week.toFixed(2).replace('.', ',')}`;
                    document.getElementById('vendas-mes').textContent = `R$ ${summary.month.toFixed(2).replace('.', ',')}`;
                    document.getElementById('pedidos-pendentes').textContent = summary.pending;
                }

            } catch (error) {
                console.error('Erro ao carregar dados:', error);
                mostrarMensagem(error.message, 'error');

                // Renderiza tabela vazia
                renderizarTabelaVendas([]);
                atualizarResumo({
                    today: 0,
                    week: 0,
                    month: 0,
                    pending: 0
                });
            }
        }

        function atualizarResumo(summary) {
            const formatarMoeda = (valor) => `R$ ${valor.toFixed(2).replace('.', ',')}`;

            document.getElementById('vendas-hoje').textContent = formatarMoeda(summary.today);
            document.getElementById('vendas-semana').textContent = formatarMoeda(summary.week);
            document.getElementById('vendas-mes').textContent = formatarMoeda(summary.month);
            document.getElementById('pedidos-pendentes').textContent = summary.pending;
        }

        // Modifique a função atualizarResumo():
        function atualizarResumo(summary) {
            document.getElementById('vendas-hoje').textContent =
                `R$ ${summary.today.toFixed(2).replace('.', ',')}`;

            document.getElementById('vendas-semana').textContent =

            switch (order) {
                case 'data-desc':
                    sortedVendas.sort((a, b) => new Date(b.order_date || b.data) - new Date(a.order_date || a.data));

                    sortedVendas.sort((a, b) => new Date(a.order_date || a.data) - new Date(b.order_date || b.data));

app.get('/api/sales', async (req, res) => {
    try {
        const connection = await pool.getConnection();

        const [orders] = await connection.query(`
            SELECT o.*, c.name as customer_name,
                o.channel,
                o.whatsapp_number,
                o.ifood_order,
                o.attendant,
                COUNT(oi.id) as items_count
            FROM orders o
            LEFT JOIN customers c ON o.customer_id = c.id
            LEFT JOIN order_items oi ON o.id = oi.order_id
            GROUP BY o.id
            ORDER BY o.order_date DESC
        `);

        const [items] = await connection.query(`
            SELECT oi.order_id, p.name as product, oi.quantity, oi.unit_price
            FROM order_items oi
            JOIN products p ON oi.product_id = p.id
        `);

        connection.release();

        const sales = orders.map(order => ({
            ...order,
            items: items.filter(item => item.order_id === order.id).map(item => ({
                product: item.product,
                quantity: item.quantity,
                valor: item.unit_price
            })),
            order_date: new Date(order.order_date).toISOString()
        }));

        res.json(sales);
    } catch (error) {
        console.error(error);
        res.status(500).json({ error: 'Database error' });
    }
});

app.get('/api/sales/summary', async (req, res) => {
    try {
        const connection = await pool.getConnection();

        const [today] = await connection.query(`
            SELECT IFNULL(SUM(total_amount), 0) as total
            FROM orders
            WHERE DATE(order_date) = CURDATE()
        `);

        const [week] = await connection.query(`
            SELECT IFNULL(SUM(total_amount), 0) as total
            FROM orders
            WHERE YEARWEEK(order_date) = YEARWEEK(CURDATE())
        `);

        const [month] = await connection.query(`
            SELECT IFNULL(SUM(total_amount), 0) as total
            FROM orders
            WHERE MONTH(order_date) = MONTH(CURDATE())
            AND YEAR(order_date) = YEAR(CURDATE())
        `);

        const [pending] = await connection.query(`
            SELECT COUNT(*) as count
            FROM orders
            WHERE status IN ('pending', 'preparing')
        `);

        connection.release();

        // Garante que os valores sejam números
        res.json({
            today: parseFloat(today[0].total) || 0,
            week: parseFloat(week[0].total) || 0,
            month: parseFloat(month[0].total) || 0,
            pending: parseInt(pending[0].count) || 0
        });
    } catch (error) {
        console.error(error);
        // Retorna valores padrão em caso de erro
        res.json({
            today: 0,
            week: 0,
            month: 0,
            pending: 0
        });
    }
});

app.post('/api/sales', async (req, res) => {
    let connection;
    try {
        const { customer, channel, items, total, status, observations, channelData } = req.body;
        connection = await pool.getConnection();
        await connection.beginTransaction();

        // 1. Criar ou buscar cliente
        const [customerResult] = await connection.query(
            `INSERT INTO customers (name) VALUES (?) 
             ON DUPLICATE KEY UPDATE id=LAST_INSERT_ID(id)`,
            [customer]
        );

        // 2. Criar a ordem com todos os campos necessários
        const [orderResult] = await connection.query(
            `INSERT INTO orders (
                customer_id, 
                total_amount, 
                status, 
                notes,
                channel,
                whatsapp_number,
                ifood_order,
                attendant
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
            [
                customerResult.insertId,
                total,
                status,
                observations || null, // Corrigido campo notes
                channel,
                channelData?.whatsapp || null,
                channelData?.ifood || null,
                channelData?.attendant || null
            ]
        );

        // 3. Inserir itens com validação de produto
        for (const item of items) {
            const [product] = await connection.query(
                `SELECT id FROM products 
                 WHERE name = ? LIMIT 1`,
                [item.product]
            );

            if (product.length > 0) {
                await connection.query(
                    `INSERT INTO order_items 
                     (order_id, product_id, quantity, unit_price)
                     VALUES (?, ?, ?, ?)`,
                    [orderResult.insertId, product[0].id, item.quantity, item.valor]
                );
            }
        }

        await connection.commit();
        res.json({ 
            success: true, 
            id: orderResult.insertId,
            message: 'Venda registrada com sucesso!'
        });

    } catch (error) {
        if (connection) {
            await connection.rollback();
            console.error('Rollback executado devido a erro:', error);
        }
        res.status(500).json({ 
            error: 'Erro ao processar venda',
            details: error.message
        });
    } finally {
        if (connection) connection.release();
    }
});

// Rota para deletar uma venda
app.delete('/api/sales/:id', async (req, res) => {
    try {
        const connection = await pool.getConnection();
        await connection.beginTransaction();

        await connection.query('DELETE FROM order_items WHERE order_id = ?', [req.params.id]);
        await connection.query('DELETE FROM orders WHERE id = ?', [req.params.id]);

        await connection.commit();
        connection.release();

        res.json({ success: true });
    } catch (error) {
        await connection.rollback();
        connection.release();
        console.error(error);
        res.status(500).json({ error: 'Database error' });
    }
});
